Aggregate Function ExampleÂ¶

Time-series aggregation function
performs a rolling aggregation on {df_artifact}, over {window} by the selected {keys} applying {metric_aggs} on {metrics} and {label_aggs} on {labels}. adding {suffix} to the feature names.

!/User/align_mlrun.sh

Both server & client are aligned (0.6.5).

import pandas as pd
import mlrun
from mlrun import import_function

# declare the dataset
DATA = "https://s3.wasabisys.com/iguazio/data/market-palce/aggregate/metrics.pq"

# import the function
aggregate_function = import_function("hub://aggregate").apply(mlrun.platforms.auto_mount())

aggregate_run = aggregate_function.run(params={'metrics': ['cpu_utilization'],
               'labels': ['is_error'],
               'metric_aggs': ['mean', 'sum'],
               'label_aggs': ['max'],
               'suffix': 'daily',
               'inplace': False,
               'window': 5,
               'center': True,               
               'files_to_select': 2}           
           , inputs={'df_artifact': DATA}
           )
